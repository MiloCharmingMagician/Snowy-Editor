.spk - snowy level pack file format
default width and height is 20x14

struct_root {
    u32 magic; // always "SH"
	u32 fixed; // value is 16 always
	u32 levelCount;
    
    struct_level {
	    u16 world; // (1 = world1, 0 = world2, 2 = world3, 3 = world4, 4 = world5-6)
	    u16 mapWidth;
		u16 mapHeight;
		
		u16 tile[mapWidth][mapHeight];
		u16 object[mapWidth][mapHeight];
	};
};
